DROP TABLE productos; para eliminar una tabla

crear tabla:
CREATE TABLE productos (
    id INTEGER PRIMARY KEY,
    nombre TEXT NOT NULL,
    precio REAL NOT NULL,
    categoria TEXT,
    stock INTEGER DEFAULT 0
);

darle valores a la tabla:

Productos con stock mayor a 20

SELECT nombre, stock
FROM productos
WHERE stock > 20;


Productos que NO sean de la categoría 'Accesorios'
SELECT nombre, categoria
FROM productos
WHERE categoria <> 'Accesorios';

Productos con precio entre 50 y 200
SELECT nombre, precio
FROM productos
WHERE precio >= 50 AND precio <= 200;


Ordenar por categoría ascendente
SELECT nombre, categoria, precio
FROM productos
ORDER BY categoria ASC;

Ordenar por stock de mayor a menor
SELECT nombre, stock
FROM productos
ORDER BY stock DESC;

Ordenar primero por categoría y dentro de cada categoría por precio ascendente
SELECT nombre, categoria, precio
FROM productos
ORDER BY categoria ASC, precio ASC;


Productos 'electrónicos' y precio > 100

SELECT nombre, precio, categoria
FROM productos
WHERE categoria = 'Electrónica'
  AND precio > 100;

Productos con stock < 20 o precio > 500
SELECT nombre, stock, precio
FROM productos
WHERE stock < 20
   OR precio > 500;


Productos de categoría ‘Audio’ o que tengan una “a” en el nombre
SELECT nombre, categoria
FROM productos
WHERE categoria = 'Audio'
   OR nombre LIKE '%a%';


Electrónica y (precio < 200 o stock > 10)
SELECT nombre, precio, stock
FROM productos
WHERE categoria = 'Electrónica'
  AND (precio < 200 OR stock > 10);
